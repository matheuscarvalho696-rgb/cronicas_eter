rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function isSelf(userId) {
      return signedIn() && request.auth.uid == userId;
    }

    function isBootstrapAdminEmail() {
      return signedIn()
        && request.auth.token.email != null
        && request.auth.token.email == 'matheuscarvalho696@gmail.com';
    }

    function isAdmin() {
      return signedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin'
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.status == 'approved';
    }

    match /settings/{docId} {
      allow read: if true;
      allow write: if isAdmin() || isBootstrapAdminEmail();
    }

    match /users/{userId} {
      // O jogador cria apenas a própria conta.
      // A conta bootstrap do Matheus pode nascer como admin.
      allow create: if isSelf(userId) && (
        (
          request.resource.data.role == 'player'
          && request.resource.data.status == 'approved'
        )
        ||
        (
          isBootstrapAdminEmail()
          && request.resource.data.role == 'admin'
          && request.resource.data.status == 'approved'
        )
      );

      allow read: if isSelf(userId) || isAdmin();

      // Usuário pode atualizar dados próprios, mas não pode se promover.
      allow update: if isSelf(userId)
        && !('role' in request.resource.data.diff(resource.data).changedKeys())
        && !('status' in request.resource.data.diff(resource.data).changedKeys())
        && !('premium' in request.resource.data.diff(resource.data).changedKeys());

      // A conta bootstrap do Matheus pode corrigir a si mesma para admin.
      allow update: if isSelf(userId)
        && isBootstrapAdminEmail()
        && request.resource.data.role == 'admin'
        && request.resource.data.status == 'approved';

      allow update, delete: if isAdmin();
    }

    match /logs/{docId} {
      allow create: if signedIn();
      allow read: if isAdmin();
      allow update, delete: if false;
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
