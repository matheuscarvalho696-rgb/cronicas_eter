rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function isSelf(userId) {
      return signedIn() && request.auth.uid == userId;
    }

    function isBootstrapAdminEmail() {
      return signedIn()
        && request.auth.token.email != null
        && request.auth.token.email == 'matheuscarvalho696@gmail.com';
    }

    function isAdmin() {
      return signedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin'
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.status == 'approved';
    }

    match /settings/{docId} {
      allow read: if true;
      allow write: if isAdmin() || isBootstrapAdminEmail();
    }

    match /inviteCodes/{codeId} {
      // Qualquer visitante pode testar um código específico, mas não pode listar todos.
      allow get: if true;
      allow list: if isAdmin();

      allow create, delete: if isAdmin();

      // Admin pode editar; usuário autenticado pode apenas marcar o próprio código como usado no cadastro.
      allow update: if isAdmin() || (
        signedIn()
        && resource.data.status == 'active'
        && request.resource.data.status == 'used'
        && request.resource.data.usedBy == request.auth.uid
      );
    }

    match /users/{userId} {
      allow create: if isSelf(userId) && (
        (
          request.resource.data.role == 'player'
          && request.resource.data.status == 'approved'
        )
        ||
        (
          isBootstrapAdminEmail()
          && request.resource.data.role == 'admin'
          && request.resource.data.status == 'approved'
        )
      );

      allow read: if isSelf(userId) || isAdmin();

      allow update: if isSelf(userId)
        && !('role' in request.resource.data.diff(resource.data).changedKeys())
        && !('status' in request.resource.data.diff(resource.data).changedKeys())
        && !('premium' in request.resource.data.diff(resource.data).changedKeys());

      allow update: if isSelf(userId)
        && isBootstrapAdminEmail()
        && request.resource.data.role == 'admin'
        && request.resource.data.status == 'approved';

      allow update, delete: if isAdmin();
    }

    match /logs/{docId} {
      allow create: if signedIn();
      allow read: if isAdmin();
      allow update, delete: if false;
    }


    match /characters/{docId} {
      allow create: if signedIn() && request.resource.data.ownerUid == request.auth.uid;
      allow read, update, delete: if isAdmin() || (signedIn() && resource.data.ownerUid == request.auth.uid);
    }

    match /distributions/{docId} {
      allow create: if signedIn() && request.resource.data.ownerUid == request.auth.uid;
      allow read, update, delete: if isAdmin() || (signedIn() && resource.data.ownerUid == request.auth.uid);
    }

    match /familiars/{docId} {
      allow create: if signedIn() && request.resource.data.ownerUid == request.auth.uid;
      allow read, update, delete: if isAdmin() || (signedIn() && resource.data.ownerUid == request.auth.uid);
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
