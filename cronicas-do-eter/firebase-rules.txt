rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function isSelf(userId) {
      return signedIn() && request.auth.uid == userId;
    }

    function isBootstrapAdminEmail() {
      return signedIn()
        && request.auth.token.email != null
        && request.auth.token.email == 'matheuscarvalho696@gmail.com';
    }

    function isAdmin() {
      return signedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin'
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.status == 'approved';
    }

    match /settings/{docId} {
      allow read: if true;
      allow write: if isAdmin() || isBootstrapAdminEmail();
    }

    match /inviteCodes/{codeId} {
      // Qualquer visitante pode testar um código específico, mas não pode listar todos.
      allow get: if true;
      allow list: if isAdmin();

      // Admin pode criar códigos manualmente. Visitantes podem criar apenas códigos gerados pelo fluxo de e-mail.
      allow create: if isAdmin() || (
        !signedIn()
        && request.resource.data.code == codeId
        && request.resource.data.status == 'active'
        && request.resource.data.source == 'email-request'
        && request.resource.data.targetEmail is string
        && request.resource.data.targetName is string
      );
      allow delete: if isAdmin();

      // Admin pode editar; usuário autenticado pode apenas marcar o próprio código como usado no cadastro.
      allow update: if isAdmin() || (
        signedIn()
        && resource.data.status == 'active'
        && request.resource.data.status == 'used'
        && request.resource.data.usedBy == request.auth.uid
      );
    }

    match /users/{userId} {
      allow create: if isSelf(userId) && (
        (
          request.resource.data.role == 'player'
          && request.resource.data.status in ['pending_email', 'pending', 'approved']
        )
        ||
        (
          isBootstrapAdminEmail()
          && request.resource.data.role == 'admin'
          && request.resource.data.status == 'approved'
        )
      );

      allow read: if isSelf(userId) || isAdmin();

      allow update: if isSelf(userId)
        && !('role' in request.resource.data.diff(resource.data).changedKeys())
        && !('status' in request.resource.data.diff(resource.data).changedKeys())
        && !('premium' in request.resource.data.diff(resource.data).changedKeys());

      // Permite que o próprio usuário finalize a validação por código de e-mail.
      allow update: if isSelf(userId)
        && resource.data.status in ['pending_email', 'pending']
        && request.resource.data.status == 'approved'
        && request.resource.data.premium == true;

      allow update: if isSelf(userId)
        && isBootstrapAdminEmail()
        && request.resource.data.role == 'admin'
        && request.resource.data.status == 'approved';

      allow update, delete: if isAdmin();
    }

    match /emailVerifications/{userId} {
      allow create, read, update, delete: if isSelf(userId) || isAdmin();
    }

    match /logs/{docId} {
      allow create: if signedIn();
      allow read: if isAdmin();
      allow update, delete: if false;
    }


    match /characters/{docId} {
      allow create: if signedIn() && request.resource.data.ownerUid == request.auth.uid;
      allow read, update, delete: if isAdmin() || (signedIn() && resource.data.ownerUid == request.auth.uid);
    }

    match /distributions/{docId} {
      allow create: if signedIn() && request.resource.data.ownerUid == request.auth.uid;
      allow read, update, delete: if isAdmin() || (signedIn() && resource.data.ownerUid == request.auth.uid);
    }

    match /familiars/{docId} {
      allow create: if signedIn() && request.resource.data.ownerUid == request.auth.uid;
      allow read, update, delete: if isAdmin() || (signedIn() && resource.data.ownerUid == request.auth.uid);
    }

    match /roll20Tables/{docId} {
      // Todos os usuários logados podem ver as mesas.
      allow read: if signedIn();

      // Apenas Admin pode criar, editar ou remover mesas.
      allow create: if isAdmin() && request.resource.data.ownerUid == request.auth.uid;
      allow update, delete: if isAdmin();
    }

    match /roll20Imports/{docId} {
      // Todos os usuários logados podem ver personagens e folhetos importados.
      allow read: if signedIn();

      // Apenas Admin pode importar, editar, ocultar ou remover conteúdos.
      allow create: if isAdmin() && request.resource.data.ownerUid == request.auth.uid;
      allow update, delete: if isAdmin();

      // Jogador logado pode reivindicar personagem da categoria Jogadores.
      allow update: if signedIn()
        && resource.data.kind == 'character'
        && resource.data.category == 'Jogadores'
        && resource.data.visible != false
        && (!('claimedByUid' in resource.data) || resource.data.claimedByUid == '' || resource.data.claimedByUid == request.auth.uid)
        && request.resource.data.claimedByUid == request.auth.uid
        && request.resource.data.diff(resource.data).changedKeys().hasOnly([
          'claimedByUid', 'claimedByEmail', 'claimedCharacterId', 'claimedAt', 'updatedAt'
        ]);
    }

    match /{document=**} {
      allow read, write: if false;
    }
  }
}
